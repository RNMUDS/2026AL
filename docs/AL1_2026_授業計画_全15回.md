### 演習課題

**演習1: 二分探索の実装**
```python
def binary_search(data, target):
    left, right = 0, len(data) - 1
    count = 0
    while left <= right:
        count += 1
        mid = (left + right) // 2
        if data[mid] == target:
            return mid, count
        elif data[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1, count
```

**演習1: ハッシュテーブルの構築**
```python
class SimpleHashTable:
    def __init__(self, size=10):
        self.size = size
        self.table = [None] * size
    
    def hash(self, key):
        return sum(ord(c) for c in key) % self.size
    
    def insert(self, key, value):
        index = self.hash(key)
        # 衝突があれば次の空きスロットへ（線形探索法）
        while self.table[index] is not None:
            if self.table[index][0] == key:
                break  # 同じキーなら上書き
            index = (index + 1) % self.size
        self.table[index] = (key, value)
```

def insertion_sort(arr):
    comparisons = 0
    for i in range(1, len(arr)):
        key = arr[i]  # 手に取ったカード
        j = i - 1
        while j >= 0 and arr[j] > key:
            comparisons += 1
            arr[j + 1] = arr[j]  # 右にずらす

        arr[j + 1] = key  # 正しい位置に挿入

### 演習課題

**演習1: カウントソートの実装**
```python
def counting_sort(arr, max_val):
    # Step 1: 度数集計
    count = [0] * (max_val + 1)
    for x in arr:
        count[x] += 1
    
    # Step 2: 累積度数
    for i in range(1, len(count)):
        count[i] += count[i - 1]
    
    # Step 3: 出力配列に配置（安定ソート版）
    output = [0] * len(arr)
    for x in reversed(arr):
        count[x] -= 1
        output[count[x]] = x
    
    return output
```

**演習1: 迷路探索（BFS）**
```python
from collections import deque

def bfs(maze, start, goal):
    queue = deque([start])
    visited = set([start])
    parent = {start: None}
    
    while queue:
        current = queue.popleft()
        if current == goal:
            # 経路を復元して返す
            path = []
            while current is not None:
                path.append(current)
                current = parent[current]
            return path[::-1]
        
        row, col = current
        for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
            nr, nc = row+dr, col+dc
            new_pos = (nr, nc)
            if (0 <= nr < len(maze) and 0 <= nc < len(maze[0])
                and maze[nr][nc] == 0 and new_pos not in visited):
                visited.add(new_pos)
                queue.append(new_pos)
                parent[new_pos] = current
    return None
```
