説明

1.1 通常の挿入ソート（復習）

前回学んだ挿入ソートでは、ソート済み部分から挿入位置を線形探索（1つずつ比較）で見つけていました。ソート済み部分が長くなると、比較回数が増えます（最悪 O(n) 回）。

概念
配列: [2, 5, 8, | 3, 7, 1]
                ↑ この3を挿入したい
→ 8と比較: 3 < 8 → 右にずらす
→ 5と比較: 3 < 5 → 右にずらす
→ 2と比較: 3 > 2 → ここに挿入!

1.2 二分挿入ソート（Binary Insertion Sort）

辞書で単語を引くとき、端から1ページずつめくりませんよね。真ん中を開いて、目的の単語が左半分か右半分か判断して、さらに半分に絞る。これが二分探索です。

アイデア: ソート済み部分はすでに整列しているので、二分探索で挿入位置を効率的に見つけられます。

  • 比較回数: O(n) → O(log n) に改善
  • ただし、要素をずらす操作は依然 O(n) なので、全体は O(n2) のまま

1.3 Python の bisect モジュール

関数説明
bisect.bisect_left(a, x, lo, hi)ソート済みリスト a で、x を挿入すべき左端の位置を返す
bisect.bisect_right(a, x, lo, hi)同上、ただし右端の位置を返す
bisect.insort_left(a, x)x をソート済みリスト a に挿入する

1.4 バブルソート（Bubble Sort）

水中の泡が上に浮かぶように、小さい値が配列の先頭に「浮かび上がる」ことからこの名前がつきました。隣同士を比較して交換を繰り返します。

最適化: あるパスで1回も交換が起きなければ、すでにソート済みなので終了できます。

1.5 シェルソート（Shell Sort）

挿入ソートの改良版で、離れた要素同士を先に比較・交換し、間隔を徐々に縮めていきます。間隔が1になったとき通常の挿入ソートになりますが、その時点でほぼ整列済みなので高速に動作します。

例題 1

bisect で挿入位置を見つける

Python
import bisect

sorted_list = [10, 20, 30, 40, 50]
target = 25
position = bisect.bisect_left(sorted_list, target)
print(f"{target} の挿入位置: {position}")  # → 2

# 重複がある場合の left と right の違い
list_with_dup = [10, 20, 30, 30, 30, 40, 50]
left_pos  = bisect.bisect_left(list_with_dup, 30)   # → 2
right_pos = bisect.bisect_right(list_with_dup, 30)  # → 5

# insort_left で実際に挿入
demo = [10, 20, 30, 40, 50]
bisect.insort_left(demo, 25)
print(f"挿入後: {demo}")  # → [10, 20, 25, 30, 40, 50]
例題 2

二分挿入ソートの実装

Python
import bisect

def binary_insertion_sort(data):
    n = len(data)
    for i in range(1, n):
        key = data[i]
        position = bisect.bisect_left(data, key, 0, i)
        print(f"  Step{i}: key={key} → 挿入位置={position}")
        for j in range(i, position, -1):
            data[j] = data[j - 1]
        data[position] = key
        print(f"         結果: {data}")
    return data

test = [38, 27, 43, 3, 9, 82, 10]
binary_insertion_sort(test[:])
例題 3

バブルソートの基本

Python
def bubble_sort(data):
    n = len(data)
    comparison_count = 0
    swap_count = 0

    for i in range(n):
        swapped = False
        for j in range(n - 1, i, -1):
            comparison_count += 1
            if data[j - 1] > data[j]:
                temp = data[j - 1]
                data[j - 1] = data[j]
                data[j] = temp
                swap_count += 1
                swapped = True
        print(f"  パス{i+1}: {data}")
        if not swapped:
            print("  交換なし → ソート完了!")
            break

    print(f"比較回数: {comparison_count}, 交換回数: {swap_count}")

bubble_sort([5, 3, 8, 1, 2])
例題 4

挿入ソートとバブルソートの比較

同じランダムデータに対して、挿入ソートとバブルソートの比較回数・実行時間を測定して比較します。

Python
import time, random

def insertion_sort_count(data):
    comparisons = 0
    for i in range(1, len(data)):
        key = data[i]; j = i - 1
        while j >= 0:
            comparisons += 1
            if data[j] > key: data[j+1] = data[j]; j -= 1
            else: break
        data[j+1] = key
    return comparisons

def bubble_sort_count(data):
    comparisons = 0
    for i in range(len(data)):
        swapped = False
        for j in range(len(data)-1, i, -1):
            comparisons += 1
            if data[j-1] > data[j]:
                data[j-1], data[j] = data[j], data[j-1]
                swapped = True
        if not swapped: break
    return comparisons

data_size = 2000
test_data = [random.randint(0, 10000) for _ in range(data_size)]

start = time.time()
ins_comps = insertion_sort_count(test_data[:])
ins_time = time.time() - start

start = time.time()
bub_comps = bubble_sort_count(test_data[:])
bub_time = time.time() - start

print(f"挿入ソート:  比較{ins_comps}回, {ins_time:.4f}秒")
print(f"バブルソート: 比較{bub_comps}回, {bub_time:.4f}秒")

# ===== 発展課題: ソートアルゴリズムの選択基準 =====

import random
import bisect

# --- 挿入ソート（比較回数・交換回数を計測） ---
def insertion_sort(data):
    arr = data[:]
    comparisons = 0
    swaps = 0
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0:
            comparisons += 1
            if arr[j] > key:
                arr[j + 1] = arr[j]
                swaps += 1
                j -= 1
            else:
                break
        arr[j + 1] = key
    return arr, comparisons, swaps

# --- 二分挿入ソート（比較回数・交換回数を計測） ---
def binary_insertion_sort(data):
    arr = data[:]
    comparisons = 0
    swaps = 0
    for i in range(1, len(arr)):
        key = arr[i]
        # 二分探索で挿入位置を決定
        lo, hi = 0, i
        while lo < hi:
            comparisons += 1
            mid = (lo + hi) // 2
            if arr[mid] > key:
                hi = mid
            else:
                lo = mid + 1
        # 要素をずらす
        for j in range(i, lo, -1):
            arr[j] = arr[j - 1]
            swaps += 1
        arr[lo] = key
    return arr, comparisons, swaps

# --- バブルソート（比較回数・交換回数を計測） ---
def bubble_sort(data):
    arr = data[:]
    comparisons = 0
    swaps = 0
    n = len(arr)
    for i in range(n - 1):
        swapped = False
        for j in range(n - 1 - i):
            comparisons += 1
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swaps += 1
                swapped = True
        if not swapped:
            break
    return arr, comparisons, swaps

# --- テストデータ生成 ---
def nearly_sorted(n):
    """ほぼソート済みデータ（数個だけランダムに入れ替え）"""
    data = list(range(n))
    for _ in range(n // 20):  # 5%の要素を入れ替え
        i, j = random.sample(range(n), 2)
        data[i], data[j] = data[j], data[i]
    return data

def random_data(n):
    """完全にランダムなデータ"""
    return [random.randint(0, n * 10) for _ in range(n)]

def streaming_data(n):
    """リアルタイム到着（ランダム順）"""
    return [random.randint(0, n * 10) for _ in range(n)]

# --- 実験実行 ---
scenarios = [
    ("1. ほぼソート済み(100件)", nearly_sorted(100)),
    ("2. 完全ランダム(1000件)", random_data(1000)),
    ("3. リアルタイム到着(100件)", streaming_data(100)),
]

algorithms = [
    ("挿入ソート", insertion_sort),
    ("二分挿入ソート", binary_insertion_sort),
    ("バブルソート", bubble_sort),
]

for scenario_name, data in scenarios:
    print(f"\n=== {scenario_name} ===")
    print(f"{'アルゴリズム':>14} | {'比較回数':>10} | {'交換回数':>10}")
    print("-" * 42)
    for algo_name, algo_func in algorithms:
        _, comp, swap = algo_func(data)
        print(f"{algo_name:>14} | {comp:>10,} | {swap:>10,}")

評価ポイント（これが説明できていればOK）:

  • ほぼソート済みデータでは挿入ソートが最も少ない比較・交換回数になる（ほぼ O(n)）
  • 二分挿入ソートは比較回数を O(n log n) に削減するが、要素の移動回数は挿入ソートと同じ
  • バブルソートは全ての場面で比較回数が多く、特にランダムデータでは最も遅い
  • データの特性（ほぼ整列済みか、ランダムか、少量か）によって最適なアルゴリズムが変わることを実験で確認できた
