コードレビューとは

コードレビューとは、他の人が書いたコードを読んで改善点を見つける作業のこと。プロのソフトウェア開発では、ほぼ全てのコードがレビューを経てから本番環境に反映されます。

例題 1

バグのあるコードを見つける（正しさ）

Python
# --- バグのあるコード ---
def find_max_buggy(numbers):
    """リストの最大値を返す関数（バグあり）"""
    maximum = numbers[0]  # 空リストでエラー！
    for i in range(len(numbers)):
        if numbers[i] > maximum:
            maximum = numbers[i]
    return maximum

# --- 修正版 ---
def find_max_fixed(numbers):
    """リストの最大値を返す関数（修正版）"""
    if not numbers:  # 空リストチェックを追加
        print("警告: 空のリストです")
        return None
    maximum = numbers[0]
    for i in range(1, len(numbers)):  # 2番目から開始
        if numbers[i] > maximum:
            maximum = numbers[i]
    return maximum
  • バグ1: 空リストが渡されると numbers[0] で IndexError
  • 修正: if not numbers: で空チェックを追加
  • 改善: ループを range(1, ...) にして無駄な比較を削除

例題 3

効率性の改善（リスト vs セット）

Python
import time

def has_duplicate_slow(data):
    """重複チェック（リスト版 = 遅い O(n^2)）"""
    seen = []  # リストの検索は O(n)
    for item in data:
        if item in seen:  # O(n) の検索が n 回 → O(n^2)
            return True
        seen.append(item)
    return False

def has_duplicate_fast(data):
    """重複チェック（セット版 = 速い O(n)）"""
    seen = set()  # セットの検索は O(1)
    for item in data:
        if item in seen:  # O(1) の検索が n 回 → O(n)
            return True
        seen.add(item)
    return False

エラーハンドリングの追加

Python
def safe_divide(numerator, denominator):
    """安全に割り算をする関数"""
    try:
        result = numerator / denominator
        return result
    except ZeroDivisionError:
        print("  エラー: 0で割ることはできません")
        return None
    except TypeError:
        print("  エラー: 数値でない値が渡されました")
        return None

def validate_maze_input(maze, start, goal):
    """迷路の入力データを検証する関数"""
    errors = []
    if not maze:
        errors.append("迷路が空です")
        return errors
    rows = len(maze)
    cols = len(maze[0])
    sr, sc = start
    if not (0 <= sr < rows and 0 <= sc < cols):
        errors.append(f"スタート{start}が範囲外です")
    elif maze[sr][sc] != 0:
        errors.append(f"スタート{start}が壁です")
    return errors
