コードのループ構造から計算量をO記法で求めます。

計算量のパターン

パターン計算量コード例
ループなしO(1)value = data[0]

Python
# --- 実測による確認 ---
sizes = [1000, 2000, 4000, 8000]

for n in sizes:
    data = list(range(n))

    # O(n) の処理
    start_time = time.perf_counter()
    total = 0
    for val in data:
        total += val
    time_n = time.perf_counter() - start_time

    # O(n^2) の処理
    start_time = time.perf_counter()
    count = 0
    for i in range(n):
        for j in range(n):
            count += 1
    time_n2 = time.perf_counter() - start_time

# nが2倍 → O(n)は約2倍、O(n^2)は約4倍になる

例題 4

同じ問題を2つのアルゴリズムで解く

「リスト内の重複要素を見つける」問題を2通りで解き、性能を比較します。

Python
def find_duplicates_bruteforce(data):
    """総当たり版 O(n^2)"""
    duplicates = []
    for i in range(len(data)):
        for j in range(i + 1, len(data)):
            if data[i] == data[j] and data[i] not in duplicates:
                duplicates.append(data[i])
    return duplicates

def find_duplicates_hashset(data):
    """ハッシュセット版 O(n)"""
    seen = set()
    duplicates = set()
    for item in data:
        if item in seen:
            duplicates.add(item)
        else:
            seen.add(item)
    return list(duplicates)

標準課題 2

総合トレース：探索＋ソート＋辞書の組み合わせ

以下のコードを実行せずに、各ステップの変数変化と最終出力を予測しなさい。
予測を書いた後に、AL15-3.py として保存・実行して確かめなさい。

Python ── AL15-3.py
data = [4, 2, 7, 1, 5, 3, 6]

# Step 1: 挿入ソート
for i in range(1, len(data)):
    key = data[i]
    j = i - 1
    while j >= 0 and data[j] > key:
        data[j + 1] = data[j]
        j -= 1
    data[j + 1] = key
print(f"ソート後: {data}")

# Step 2: 二分探索
target = 5
left, right = 0, len(data) - 1
count = 0
while left <= right:
    mid = (left + right) // 2
    count += 1
    if data[mid] == target:
        break
    elif data[mid] < target:
        left = mid + 1
    else:
        right = mid - 1
print(f"target={target}: index={mid}, 比較{count}回")

# Step 3: 辞書で集計
groups = {}
for d in data:
    key = "偶数" if d % 2 == 0 else "奇数"
    if key not in groups:
        groups[key] = []
    groups[key].append(d)
print(f"分類: {groups}")

ヒント: 3つのステップを順番に追います。Step 1 のソート結果が Step 2 の入力になります。Step 3 は独立した処理です。

# ===== 発展課題: 学期の振り返りデモ =====
# 例: 二分探索のデモプログラム
# （学生は自分が最も印象に残ったアルゴリズムを選んでください）

import random
import time

def demo_binary_search():
    """二分探索の動作を視覚的にデモする"""
    # ソート済みデータを準備
    data = sorted(random.sample(range(1, 1000), 50))
    target = random.choice(data)  # 必ず見つかる値を選ぶ

    print("=" * 60)
    print("  二分探索デモ: 50個のソート済みデータから値を探す")
    print("=" * 60)
    print(f"\nデータ（先頭10個）: {data[:10]} ...")
    print(f"探す値: {target}")
    print()

    low = 0
    high = len(data) - 1
    step = 0

    while low <= high:
        step += 1
        mid = (low + high) // 2

        # 現在の探索範囲を表示
        range_str = ""
        for i in range(len(data)):
            if i == mid:
                range_str += f"[{data[i]}]"  # 中央値
            elif low <= i <= high:
                range_str += f" {data[i]} "  # 探索範囲内
            else:
                range_str += " .. "  # 範囲外

        print(f"ステップ{step}: 範囲=[{data[low]}〜{data[high]}]"
              f"({high - low + 1}個), 中央値={data[mid]}", end="")

        if data[mid] == target:
            print(f" → 発見! インデックス={mid}")
            break
        elif data[mid] < target:
            print(f" → {data[mid]} < {target} なので右半分へ")
            low = mid + 1
        else:
            print(f" → {data[mid]} > {target} なので左半分へ")
            high = mid - 1

    print(f"\n合計 {step} ステップで発見（最大 log2(50) ≒ 6 ステップ）")

    # 逐次探索との比較
    print("\n--- 逐次探索との比較 ---")
    linear_count = 0
    for val in data:
        linear_count += 1
        if val == target:
            break
    print(f"逐次探索: {linear_count}回の比較が必要")
    print(f"二分探索: {step}回の比較で済んだ")
    print(f"→ 二分探索は {linear_count / step:.1f}倍 高速!")

# --- デモ実行 ---
demo_binary_search()

print()
