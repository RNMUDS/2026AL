説明: bisect と応用探索

なぜ bisect を使うのか？

前回、二分探索を自分で実装しました。しかし、Pythonには標準ライブラリに二分探索が用意されています。

項目自作 binary_searchbisect モジュール
実装自分で書く標準ライブラリ
速度Python速度C言語で実装済み（高速）
機能値の検索のみ検索 + 挿入位置の特定
重複対応工夫が必要bisect_left / bisect_right で対応

bisect の主な関数

関数説明
bisect_left(a, x)x を挿入すべき左端の位置を返す
bisect_right(a, x)x を挿入すべき右端の位置を返す
insort_left(a, x)x をソートを保ったまま左端に挿入する
insort_right(a, x)x をソートを保ったまま右端に挿入する

bisect_left と bisect_right の違い

重複がない場合は同じ結果ですが、重複がある場合に差が出ます。

探索アルゴリズムの使い分け

アルゴリズム計算量条件使いどころ
逐次探索O(n)なし小データ、未ソート
二分探索O(log n)ソート済み大データの検索
bisectO(log n)ソート済みPython実用コード
ジャンプ探索O(sqrt(n))ソート済みブロック単位の探索
指数探索O(log n)ソート済み無限/巨大リスト

例題 1

bisect_left と bisect_right の違い

Python
import bisect

# --- 重複なしのデータ ---
data = [1, 3, 5, 7, 9, 11, 13]
target = 5
left_pos = bisect.bisect_left(data, target)    # 2
right_pos = bisect.bisect_right(data, target)   # 3

# --- 存在しない値 ---
target = 8
pos = bisect.bisect_left(data, target)
print(f"8を挿入するなら位置{pos}")  # 位置4（9の前）

# --- 重複ありのデータ ---
data_dup = [1, 3, 5, 5, 5, 7, 9]
target = 5
left_pos = bisect.bisect_left(data_dup, target)   # 2（最初の5の位置）
right_pos = bisect.bisect_right(data_dup, target)  # 5（最後の5の次）
print(f"5の個数: {right_pos - left_pos}個")         # 3個
print(f"5のスライス: {data_dup[left_pos:right_pos]}") # [5, 5, 5]
例題 2

insort でソート済みリストを維持する

バラバラの順番で値が到着しても、insort を使えばソートを保ったまま挿入できます。

Python
import bisect

sorted_list = []
values = [7, 3, 11, 1, 9, 5, 13, 2, 8]

for value in values:
    insert_pos = bisect.bisect_left(sorted_list, value)
    bisect.insort(sorted_list, value)
    print(f"  insort({value:>2}) → 位置{insert_pos}に挿入 → {sorted_list}")

print(f"\n最終結果: {sorted_list}")
# [1, 2, 3, 5, 7, 8, 9, 11, 13]  ← ソート済み!

ジャンプ探索

sqrt(n) ずつジャンプしてブロックを特定し、ブロック内を逐次探索します。

Python
import math

def jump_search(data, target):
    """ジャンプ探索: sqrt(n)ずつジャンプ → ブロック内を逐次探索"""
    n = len(data)
    jump_size = int(math.sqrt(n))
    print(f"  データ数: {n}, ジャンプ幅: {jump_size}")

    # ステップ1: ジャンプしてブロックを特定する
    prev = 0
    curr = 0
    while curr < n and data[curr] < target:
        prev = curr
        curr += jump_size
        print(f"  ジャンプ: 位置{curr}をチェック")

    # ステップ2: ブロック内を逐次探索する
    print(f"  ブロック内探索: 位置{prev}〜{min(curr, n-1)}")
    for i in range(prev, min(curr + 1, n)):
        if data[i] == target:
            return i
    return -1

# --- テスト ---
data = list(range(0, 100, 3))  # [0, 3, 6, 9, ..., 99]
result = jump_search(data, 42)
print(f"  結果: インデックス{result}")

指数探索

指数的に範囲を広げてから、その範囲で二分探索します。巨大なリストに有効です。

Python
def exponential_search(data, target):
    """指数探索: 指数的に範囲を広げてから二分探索"""
    n = len(data)
    if n == 0:
        return -1
    if data[0] == target:
        return 0

    # ステップ1: 指数的にインデックスを広げる
    bound = 1
    while bound < n and data[bound] < target:
        print(f"    bound={bound}, data[{bound}]={data[bound]}")
        bound *= 2   # 1 → 2 → 4 → 8 → 16 → ...

    # ステップ2: 絞り込んだ範囲で二分探索
    left = bound // 2
    right = min(bound, n - 1)
    print(f"  二分探索: [{left}..{right}]")

    while left <= right:
        mid = (left + right) // 2
        if data[mid] == target:
            return mid
        elif data[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

data = list(range(0, 200, 2))  # [0, 2, 4, ..., 198]
result = exponential_search(data, 42)
print(f"  結果: インデックス{result}")
例題 5

複数の探索アルゴリズムの性能比較

10万件のデータで逐次探索・二分探索・bisect探索の実行時間を比較します。

Python
import time, random, bisect

def linear_search(data, target):
    for i in range(len(data)):
        if data[i] == target: return i
    return -1

def binary_search(data, target):
    left, right = 0, len(data) - 1
    while left <= right:
        mid = (left + right) // 2
        if data[mid] == target: return mid
        elif data[mid] < target: left = mid + 1
        else: right = mid - 1
    return -1

def bisect_search(data, target):
    idx = bisect.bisect_left(data, target)
    if idx < len(data) and data[idx] == target: return idx
    return -1

# --- ベンチマーク ---
n = 100000
data = list(range(n))
targets = [random.randint(0, n-1) for _ in range(500)]

algorithms = [
    ("Linear Search", linear_search),
    ("Binary Search", binary_search),
    ("Bisect Search", bisect_search),
]

results = []
for name, func in algorithms:
    start = time.perf_counter()
    for t in targets: func(data, t)
    elapsed = time.perf_counter() - start
    results.append((name, elapsed))

results.sort(key=lambda x: x[1])
for rank, (name, elapsed) in enumerate(results, 1):
    print(f"  {rank}位: {name:<18} {elapsed:.4f}秒")

模範コード（AL05-4.py）:

# ===== 発展課題: ジャンプ探索と二分探索の比較 =====

import math
import random

# --- ジャンプ探索（比較回数を計測） ---
def jump_search(data, target):
    """ジャンプ探索: (見つかったか, 比較回数) を返す"""
    n = len(data)
    step = int(math.sqrt(n))  # ジャンプ幅 = sqrt(n)
    count = 0  # 比較回数

    # ステップ1: ブロック単位でジャンプ
    prev = 0
    while prev < n and data[min(prev + step, n) - 1] < target:
        count += 1
        prev += step
        if prev >= n:
            return False, count

    # ステップ2: ブロック内を逐次探索
    for i in range(prev, min(prev + step, n)):
        count += 1
        if data[i] == target:
            return True, count

    return False, count

# --- 二分探索（比較回数を計測） ---
def binary_search(data, target):
    """二分探索: (見つかったか, 比較回数) を返す"""
    low = 0
    high = len(data) - 1
    count = 0

    while low <= high:
        mid = (low + high) // 2
        count += 1
        if data[mid] == target:
            return True, count
        elif data[mid] < target:
            low = mid + 1
        else:
            high = mid - 1

    return False, count

# --- 実験 ---
sizes = [100, 10000, 1000000]
trials = 100  # 各サイズで何回探索するか

print(f"{'データサイズ':>12} | {'ジャンプ探索':>12} | {'二分探索':>10} | {'理論値 sqrt(n)':>14} | {'理論値 log2(n)':>14}")
print("-" * 75)

for n in sizes:
    data = list(range(n))  # 0〜n-1のソート済みリスト

    total_jump = 0
    total_binary = 0

    for _ in range(trials):
        target = random.randint(0, n - 1)
        _, jcount = jump_search(data, target)
        _, bcount = binary_search(data, target)
        total_jump += jcount
        total_binary += bcount

    avg_jump = total_jump / trials
    avg_binary = total_binary / trials
    theory_jump = math.sqrt(n)
    theory_binary = math.log2(n)

    print(f"{n:>12,} | {avg_jump:>12.1f} | {avg_binary:>10.1f} | {theory_jump:>14.1f} | {theory_binary:>14.1f}")

評価ポイント（これが説明できていればOK）:

  • ジャンプ探索は sqrt(n) 幅でブロックを飛ばし、該当ブロック内で逐次探索する2段階の方法
  • 二分探索 O(log n) はジャンプ探索 O(sqrt(n)) より常に比較回数が少ない（n が大きいほど差が開く）
  • データサイズが大きくなると差が顕著になる（例: n=1,000,000 で sqrt(n)=1000 vs log2(n)=20）
  • ジャンプ探索は実装が単純で、リンクリストのように「前方にしか進めない」データ構造で有利になる場合がある
