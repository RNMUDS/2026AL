アイテム収集迷路ゲーム

この回では、これまで学んだアルゴリズムを組み合わせて、迷路の中でアイテムを全て集めてゴールする最短経路を求めるプログラムを段階的に作ります。

基本の考え方

普通のBFSでは「位置」だけを状態として記録しますが、アイテム収集では「位置 + 集めたアイテムの組み合わせ」を状態として管理する必要があります。同じ位置でも、集めたアイテムが違えば別の状態として扱います。

スーパーで買い物をするとき、買い物リストの商品を全て集めてからレジに行くのと似ています。最短ルートで全商品を集めてレジ（ゴール）に行きたい。同じ通路にいても、カゴの中身が違えば「別の状態」です。

    rows = len(maze)

例題 2

BFSで最短経路を求める

Python
from collections import deque

def find_shortest_path(maze, start, goal):
    """BFSで最短経路を求める関数"""
    rows = len(maze)
    cols = len(maze[0])
    directions = [(-1,0), (1,0), (0,-1), (0,1)]

    queue = deque([start])
    visited = {start}
    parent = {start: None}

    while queue:
        current = queue.popleft()
        if current == goal:
            path = []
            node = current
            while node is not None:
                path.append(node)
                node = parent[node]
            return path[::-1]

        row, col = current
        for dr, dc in directions:
            nr, nc = row + dr, col + dc
            next_pos = (nr, nc)
            if (0 <= nr < rows and 0 <= nc < cols
                    and maze[nr][nc] == 0 and next_pos not in visited):
                visited.add(next_pos)
                queue.append(next_pos)
                parent[next_pos] = current
    return None

def check_item_pickup(path, items):
    """経路上で拾えるアイテムを調べる関数"""
    collected = []
    for pos in path:
        if pos in items and pos not in collected:
            collected.append(pos)
            print(f"  位置{pos}でアイテムを拾った！")
    return collected

例題 4

frozensetで状態を管理する

frozensetを使い、「位置 + 集めたアイテム」を1つの状態として管理する方法を学びます。

Python
# setは辞書のキーに使えないが、frozensetは使える
state_a = ((0, 0), frozenset())              # アイテム0個
state_b = ((0, 0), frozenset({(2, 0)}))       # アイテム1個
state_c = ((0, 0), frozenset({(2, 0), (0, 4)}))  # アイテム2個

visited = {}
visited[state_a] = True  # 同じ位置でも
visited[state_b] = True  # 別の状態として
visited[state_c] = True  # 管理できる！

# アイテム追加の操作
collected = frozenset()
collected = collected | {(2, 0)}  # 1個取得: frozenset({(2, 0)})
collected = collected | {(0, 4)}  # 2個取得: frozenset({(2, 0), (0, 4)})
frozenset のポイント set は変更できるので辞書のキーに使えない。frozenset は変更不可なのでキーに使える。アイテムの「持ち物セット」を管理するのに最適。
例題 5

全アイテム収集BFS（完全版）

位置とアイテム収集状態を組み合わせたBFSで、全アイテムを集めてゴールする最短経路を求めます。

Python
from collections import deque

def solve_item_maze(maze, start, goal, items):
    """全アイテムを集めてゴールする最短経路を求める関数"""
    rows = len(maze)
    cols = len(maze[0])
    directions = [(-1,0), (1,0), (0,-1), (0,1)]
    all_items = frozenset(items)

    # --- 初期状態: (位置, 空のアイテム集合) ---
    initial_state = (start, frozenset())
    queue = deque([initial_state])
    visited = {initial_state}
    parent = {initial_state: None}

    while queue:
        current_pos, collected = queue.popleft()

        # --- ゴール判定: 位置がゴール AND 全アイテム収集済み ---
        if current_pos == goal and collected == all_items:
            path = []
            state = (current_pos, collected)
            while state is not None:
                path.append(state)
                state = parent[state]
            return path[::-1]

        # --- 4方向を探索 ---
        row, col = current_pos
        for dr, dc in directions:
            nr, nc = row + dr, col + dc
            next_pos = (nr, nc)

            if (0 <= nr < rows and 0 <= nc < cols
                    and maze[nr][nc] == 0):
                new_collected = collected
                if next_pos in items:
                    new_collected = collected | {next_pos}  # アイテム取得

                new_state = (next_pos, new_collected)
                if new_state not in visited:
                    visited.add(new_state)
                    queue.append(new_state)
                    parent[new_state] = (current_pos, collected)

    return None

実行結果のイメージ

5x5の迷路でアイテム3個の場合、全アイテムを回収してゴールする最短経路は16歩。単純な最短経路（8歩）よりも長くなりますが、これが全アイテムを集める最短の方法です。

Python ── AL13-3.py
from collections import deque

maze = [
    ["S", ".", "*"],
    [".", "#", "."],
    [".", ".", "G"]
]

queue = deque([(0, 0, False, 0)])  # (row, col, アイテム取得済み, 歩数)
visited = set()
visited.add((0, 0, False))

while queue:
    r, c, has_item, steps = queue.popleft()
    if maze[r][c] == "G" and has_item:
        print(f"ゴール！ 歩数={steps}")
        break
    for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
        nr, nc = r + dr, c + dc
        if 0 <= nr < 3 and 0 <= nc < 3 and maze[nr][nc] != "#":
            new_item = has_item or (maze[nr][nc] == "*")
            if (nr, nc, new_item) not in visited:
                visited.add((nr, nc, new_item))
                queue.append((nr, nc, new_item, steps + 1))

ヒント: 状態は (行, 列, アイテム取得済みか) の3つ組です。アイテム(*) を取る前と後で別の状態になることに注意。ゴール到達時に has_item が True でなければ終了しません。

# ===== 発展課題: 複数アイテムの最適収集順序 =====

from collections import deque

# --- 迷路の定義 ---
maze = [
    "S . . . .",
    ". # # . .",
    ". . 1 . #",
    "# . # . .",
    ". 2 . 3 G",
]

# 迷路をパースする
def parse_maze(maze_lines):
    grid = []
    start = None
    goal = None
    items = {}
    for r, line in enumerate(maze_lines):
        row = line.split()
        grid.append(row)
        for c, cell in enumerate(row):
            if cell == "S":
                start = (r, c)
            elif cell == "G":
                goal = (r, c)
            elif cell.isdigit():
                items[int(cell)] = (r, c)
    return grid, start, goal, items

grid, start, goal, items = parse_maze(maze)
rows = len(grid)
cols = len(grid[0])
num_items = len(items)

print(f"迷路サイズ: {rows}x{cols}")
print(f"アイテム数: {num_items}")
print(f"アイテム位置: {items}")
print()

# --- BFSで全アイテム収集 + ゴール到達 ---
# 状態: (行, 列, 収集済みアイテムのビットマスク)
# ビットマスク: アイテム1=bit0, アイテム2=bit1, アイテム3=bit2
# 全収集 = (1 << num_items) - 1 = 0b111 = 7

all_collected = (1 << num_items) - 1

def bfs_collect(grid, start, goal, items):
    """BFSで全アイテム収集して最短でゴールする経路を探す"""
    # 初期状態
    init_state = (start[0], start[1], 0)  # (行, 列, 収集ビット)
    queue = deque([(init_state, [start])])  # (状態, 経路)
    visited = {init_state}
    max_queue = 0
    states_explored = 0

    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]  # 上下左右

    while queue:
        max_queue = max(max_queue, len(queue))
        (r, c, collected), path = queue.popleft()
        states_explored += 1

        # ゴール判定: 全アイテム収集済み かつ ゴール地点
        if (r, c) == goal and collected == all_collected:
            return path, states_explored, max_queue

        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] != "#":
                new_collected = collected
                # アイテムがある場合、収集
                for item_id, (ir, ic) in items.items():
                    if (nr, nc) == (ir, ic):
                        new_collected = collected | (1 << (item_id - 1))

                new_state = (nr, nc, new_collected)
                if new_state not in visited:
                    visited.add(new_state)
                    queue.append((new_state, path + [(nr, nc)]))

    return None, states_explored, max_queue

path, explored, max_q = bfs_collect(grid, start, goal, items)

if path:
    print(f"最短経路の長さ: {len(path) - 1}歩")
    print(f"経路: {path}")
    print(f"探索した状態数: {explored}")
    print(f"キュー最大サイズ: {max_q}")
    print(f"可能な状態数の上限: {rows} x {cols} x {2**num_items} = {rows * cols * (2**num_items)}")
else:
    print("経路が見つかりませんでした")

# --- アイテム数と状態数の関係 ---
print()
print("--- アイテム数と状態空間の増加 ---")
for n in range(1, 6):
    states = rows * cols * (2 ** n)
    print(f"  アイテム{n}個: 最大 {states:,} 状態")

評価ポイント（これが説明できていればOK）:

  • 状態を (位置, 収集済みアイテム集合) で表現し、ビットマスクで収集状態を管理している
  • アイテムが n 個の場合、状態数は「マスの数 x 2^n」に膨らむ（指数的増加）
  • BFSは全状態を探索するため、アイテム数が増えるとメモリと時間が急激に増加する
  • 現実的にはアイテム10〜15個程度が上限（2^15 = 32768）で、それ以上は別のアルゴリズムが必要
